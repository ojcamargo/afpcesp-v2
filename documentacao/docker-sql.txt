#efetuar download imagem sql server
docker pull mcr.microsoft.com/mssql/server:2022-latest

#instalar e executar imagem sql server
#atencao para senha do user SA, deve seguir as politicas de senhas da microsoft
#senhas que estejam fora das diretrizes podem causar falha de conex√£o

   -d \
   mcr.microsoft.com/mssql/server:2022-latest

#testar conexao
